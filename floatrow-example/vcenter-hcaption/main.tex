\documentclass[12pt]{ctexart}
\usepackage{graphicx}
\usepackage{caption,floatrow}

\DeclareCaptionSubType[alph]{figure}
\captionsetup[figure]{labelsep=colon}
\captionsetup[subfigure]{labelformat=brace,labelsep=space,labelfont=bf}


\begin{document}
\thisfloatsetup{heightadjust=all,valign=t}
\begin{figure}
  \ffigbox[\FBwidth]{%
    \begin{subfloatrow}
      \ffigbox[\FBwidth]
      {\includegraphics[width=3cm,height=5cm]{example-image-b}}
      {\caption{子图a}\label{sfig:testa}}%
      \ffigbox%[\FBwidth]
      {\caption{子图b}\label{sfig:testb}}
      {\includegraphics[width=3cm,height=2cm]{example-image-a}}
    \end{subfloatrow}
  }{\caption{两个图顶端对齐}}
\end{figure}
