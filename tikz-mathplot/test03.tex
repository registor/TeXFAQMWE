\usepackage{pgfplots}
\pgfplotsset{compat=newest}

\begin{document}
\begin{tikzpicture}
  \draw[scale=0.5,domain=-3:3, smooth, variable=\x,blue] plot ({\x}, {\x*\x});
  \draw[scale=0.5,domain=-3:3, smooth, variable=\y,red] plot ({\y*\y}, {\y});
  \draw[->] (-3,0) -- (4.2,0) node[right] {$x$};
  \draw[->] (0,-3) -- (0,4.2) node[above] {$y$};
\end{tikzpicture}
\end{document}
