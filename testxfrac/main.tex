	\begin{equation}
		(4 / 9)(5 / v) = (\nicefrac{4}{9})(\nicefrac{5}{v}) = \frac{2/3}{1/5} = \frac{\nicefrac{2}{3}}{\nicefrac{1}{5}}.
	\end{equation}
Ou encore
	\begin{equation}
		(4 / 9)(5 / v) = (\sfrac{4}{9})(\sfrac{5}{v}) = \frac{2/3}{1/5} = \frac{\sfrac{2}{3}}{\sfrac{1}{5}}.
	\end{equation}
