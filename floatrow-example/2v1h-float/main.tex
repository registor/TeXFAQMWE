    {\includegraphics[width=3cm,height=3cm]{example-image-a}}\vss
  \ffigbox[\FBwidth]
    {\caption{small subfigure B}\label{sfig:testb}}
    {\includegraphics[width=3cm,height=2cm]{example-image-b}}
  }
\end{subfloatrow}\hspace*{\columnsep}
\begin{subfloatrow}
  \ffigbox[\FBwidth][]

\end{subfloatrow}
