      \begin{algorithm}[H]
        \SetAlgoLined
        \LinesNumbered          
        \SetKwInput{KwIn}{输入}
        \SetKwInput{KwOut}{输出}
        \KwIn{$x_i=\{1,\nicefrac{1}{2},\nicefrac{1}{3},\nicefrac{1}{4},\nicefrac{1}{5},\cdots,\nicefrac{1}{100}\}$}
        \KwOut{$sum=\sum\limits_{i=1}^{100}  x_i$}              
        \BlankLine 
        初始化$sum=0$\;
        \For{$i\leftarrow1$ \KwTo $100$}{
          $sum=sum+\nicefrac{1}{i}$\;
          $i=i+1$\;
        }
        输出$sum$\;
        \caption{计算分数累加和}
      \end{algorithm}
